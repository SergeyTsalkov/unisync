  local: /path/to/stuff

  host: [ip]

  method: ssh | nativessh | directtls
  prefer: local | remote | newer | older (default: newer)
  cache: filename (default: [name].cache)
}

  unisync ~/localpath user@host:remotepath/
  unisync myhost.json
